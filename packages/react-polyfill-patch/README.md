# `react-polyfill-patch`

> Polyfills patch for React new api

## Support

`>= react@0.14.9`

## Usage

```js
import 'react-polyfill-patch';
import {
  // memo
  memo,
  // ref
  createRef,
  forwardRef,
  // Fragment
  Fragment,
  // PureComponent
  PureComponent,
  // context
  createContext,
  // hooks
  useRef,
  useState,
  useContext,
  useReducer,
  useEffect,
  useLayoutEffect,
  useMemo,
  useCallback,
  useImperativeHandle
} from 'react';
import { createPortal } from 'react-dom';
```
