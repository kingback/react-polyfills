# `react-polyfill-patch`

> Polyfills patch for React new api

## Support

`>= react@0.14.9`

## Usage

```js
import 'react-polyfill-patch';
import {
  // ref
  createRef,
  forwardRef,
  // context
  createContext,
  // hooks
  useRef,
  useState,
  useContext,
  useReducer,
  useEffect,
  useLayoutEffect,
  useMemo,
  useCallback,
  useImperativeHandle
} from 'react';
```
